سیگنال‌ها (ESP32 → TB6612):
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
چپ IO12 (3) ──────────────────► چپ PWMA (1)
چپ IO13 (4) ──────────────────► چپ AIN1 (3)
چپ IO15 (5) ──────────────────► چپ AIN2 (2)
چپ IO2  (7) ──────────────────► چپ BIN1 (5)
راست UOT (7) ─────────────────► چپ BIN2 (6)
چپ IO14 (6) ──────────────────► چپ PWMB (7)

تغذیه ESP32:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
منبع1 (+) ────────────────────► ESP32 چپ 5V (1)
منبع1 (-) ────────────────────► ESP32 چپ GND (2)

اتصال ESP32 به TB6612:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
ESP32 راست 3V3 (1) ───────────► TB6612 راست VCC (2)
ESP32 چپ 5V (1) ──────────────► TB6612 چپ STBY (4) ⭐
ESP32 چپ GND (2) ─────────────► TB6612 چپ GND (8)

تغذیه موتور:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
باتری (+) ────────────────────► TB6612 راست VM (1)
باتری (-) ────────────────────► TB6612 راست GND (3)
باتری (-) ────────────────────► ESP32 چپ GND (2) ⭐⭐

موتورها:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
موتور چپ (+) ─────────────────► TB6612 راست AO1 (4)
موتور چپ (-) ─────────────────► TB6612 راست AO2 (5)
موتور راست (+) ──────────────► TB6612 راست BO1 (7)
موتور راست (-) ──────────────► TB6612 راست BO2 (6)

═══════════════════════════════════════════════════════════════════
                  ⚠️ نکات بسیار مهم (حفظ کنید!)
═══════════════════════════════════════════════════════════════════

1️⃣  STBY حتماً به 5V وصل باشد
    بدون این، موتورها کار نمی‌کنند!

2️⃣  همه GND ها مشترک باشند
    ESP32 GND = منبع1 GND = باتری GND = TB6612 GND

3️⃣  GPIO 4 استفاده نشده
    GPIO 4 برای Flash LED است (نباید استفاده شود)

4️⃣  UOT همان GPIO 1 است
    Serial Monitor کار نمی‌کند (عادی است)

5️⃣  اگر موتور برعکس چرخید
    سیم‌های O1 و O2 آن موتور را جابجا کنید


═══════════════════════════════════════════════════════════════════
                      🚀 تست و راه‌اندازی
═══════════════════════════════════════════════════════════════════

1. همه اتصالات را چک کنید
2. کد را آپلود کنید (با SSID و Password درست)
3. منبع 5V ESP32 را روشن کنید
4. از مرورگر به IP وصل شوید
5. باتری موتور را وصل کنید
6. با سرعت 100 تست کنید
7. جوی استیک‌ها را امتحان کنید
8. موفق باشید! 🎉
